\begin{problem}
    ضریب عبارت
    $x^4$
    را در بسط
    $(1+x)^{-3}$
    بیابید.

    \solution{
        $$(1+x)^{-n} = \sum\limits_{k=0}^\infty {-n \choose k} x^k = \sum\limits_{k=0}^\infty (-1)^k {n+k-1 \choose k} x^k$$
        بنابراین، پاسخ مسئله برابر است با:
        $$(-1)^4 {3+4-1 \choose 4} = {6 \choose 4} = 15$$
    }
\end{problem}
