\begin{problem}
    اگر ۱۰ نوع میوه داشته باشیم، به چند روش می‌توان ۱۶ عدد میوه انتخاب کرد به طوری که از هر نوع میوه به تعداد زوج انتخاب کرده باشیم؟

    \problemsolution{
        اگر 
        $n$
        تعداد نوع میوه‌هایی باشد که انتخاب کرده‌ایم، ضریب جمله‌ی 
        $x^{16}$
        در بسط 
        $(1 + x^{2} + x^{4} + ...)^{n}$
        با تعداد روش‌های انتخاب ۱۶ عدد میوه به طوری که از
        $n$
        نوع میوه انتخاب کرده باشیم، برابر است. به این ترتیب حاصل جمع این عدد در 
        $n$
        ‌های مختلف با تعداد کل روش‌ها برابر است. با توجه به این که تعداد نوع میوه‌ها محدود است، داریم: 
        $n \leq 10$
        . به این ترتیب ضریب 
        $x^{16}$
        در عبارت زیر با تعداد کل روش‌های انتخاب ۱۶ عدد میوه از ۱۰ نوع میوه برابر است:
        \begin{center}
            $1 + (1 + x^{2} + x^{4} + ...) + (1 + x^{2} + x^{4} + ...)^{2} + ... + 
            (1 + x^{2} + x^{4} + ...)^{10}$
        \end{center}
    }
\end{problem}
