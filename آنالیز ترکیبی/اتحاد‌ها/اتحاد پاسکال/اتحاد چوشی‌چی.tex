\SUBSECTION{اتحاد چوشی‌چی}

\begin{fact}
  فرض کنیم
  $n,r$
  اعداد صحیح نامنفی باشند و
  $r \leq n$،
  آنگاه طبق تعمیم اتحاد پاسکال که با نام
  \focused{اتحاد چوشی‌چی}
  نیز شناخته می‌شود، داریم :
  \begin{center}
    $\binom{n+1}{r+1} = \sum\limits_{j=r}^n \binom{j}{r}$
  \end{center}
\end{fact}

\NOTE{
  پیش‌تر این اتحاد را در بخش دوگانه شماری، بدون استفاده از اتحاد پاسکال اثبات کرده بودیم.
}
