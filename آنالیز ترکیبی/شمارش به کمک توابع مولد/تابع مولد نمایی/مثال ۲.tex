\begin{problem}
    تعداد رشته‌های به طول $n$ از حروف 
    $a, b, c$
    که در آن‌ها تعداد $a$ و تعداد $b$ فرد باشند را بیابید. 

    \problemsolution{
        برای هر حرف یک تابع مولد نمایی در نظر می‌گیریم
        
        برای حرف $c$ محدودیتی نداریم
        
        \begin{center}
            $C(x) = \sum_{n = 0}^{\infty} \frac{x^n}{n!} = e^x$
        \end{center}
        برای حرف $a$ و $b$ که باید فرد بار آمده باشند
        
        \begin{center}
            $A(x) = B(x) = \sum_{n = 0}^{\infty} \frac{x^{2n + 1}}{(2n + 1)!}
            = \frac{e^x - e^{-x}}{2}
            = sinh(x)$
        \end{center}           
        تابع مولد نهایی برابر است با
        
        \begin{center}  
            $A(x)B(x)C(x) = (\frac{e^x - e^{-x}}{2})(\frac{e^x - e^{-x}}{2})e^x
            = (\frac{e^x - e^{-x}}{2})^2e^x
            = (\frac{e^{2x} - 2e^{x - x} + e^{-2x}}{4})e^x
            = \frac{e^{3x} - 2e^x + e^{-x}}{4}
            = \frac{1}{4}\sum_{n = 0}^{\infty} \frac{(3x)^n}{n!} - \frac{1}{2}\sum_{n = 0}^{\infty} \frac{x^n}{n!} + \frac{1}{4}\sum_{n = 0}^{\infty}\frac{(-1)^nx^n}{n!}
            = \sum_{n = 0}^{\infty} (\frac{3^n - 2 + (-1)^n}{4})\frac{x^n}{n!}$
            
        \end{center}
        بنابراین پاسخ نهایی برابر خواهد بود با:
        \begin{center}
            $\frac{3^n - 2 + (-1)^n}{4}$
        \end{center}
    }
\end{problem}
