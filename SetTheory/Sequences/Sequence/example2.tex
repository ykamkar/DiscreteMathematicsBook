\begin{problem}
    \p
مجموعه
$S$
به صورت زیر مشخص شده است. جمله عمومی آن را به دست آورید.
	\p
الف)
$1 \in S$
	\p
ب)
$\forall k \in \mathbb{N} : k \in S \Rightarrow (k + 4) \in S$
	\p
ج)هیچ عضوی در
$S$
وجود ندارد که از قانون (الف) یا (ب) به دست نیامده باشد.
    \problemsolution{
        \p
       \begin{center}
       		$1 \in S \Rightarrow 5 \in S \Rightarrow 9 \in S \Rightarrow \ldots$
       	\p
       	$S = \{1, 5, 9, 13, 17, \ldots\}$ 
        \end{center}
        \p
       جمله عمومی:
       \p
       \begin{center}
       $a_1 = 1$
       \p
       $a_2 = a_1 + 4$
       \p
       $a_3 = a_2 + 4$
       \p
       $\vdots$ 
       \p
       $a_n = a_{n-1} + 4$
       \p
       $\Downarrow$
       \p
       $a_n = 1 + 4(m - 1) = 4n - 3$
       \p
       \end{center}
        }
\end{problem}
