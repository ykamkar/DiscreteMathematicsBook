\begin{problem}
    \p
جمله عمومی تعداد پاره‌‌خط‌ها‌ی شکل زیر را به دست آورید.
	\p
    \begin{center}
     	\includegraphics[scale=0.3]{./2.jpg}
    \end{center}
    \problemsolution{
    \p
       	$$a_1 = 2$$
       	$$a_2 = a_1 + (1\times 2 + 1)$$
        $$a_3 = a_2 + (3\times 2 + 2)$$
		$$\vdots$$
		$$a_n = a_{n-1} + (n\times 2 + (n-1))$$
		$$\Downarrow$$
		$$a_n = a_1 + 2(2 + 3 + 4 +\cdots+ n) + 1 + 2 + \cdots + (n-1)$$
		$$a_1 + 2(\frac{n(n + 1)}{2} - 1) + \frac{(n - 1)n}{2}$$
		$$a_1 = 2 \rightarrow a_n = n((n + 1) + \frac{n -1}{2})$$
		$$a_n = \frac{n(3n + 1)}{2}$$
        }
\end{problem}
