\begin{problem}
    \p
    فرم بسته و ساده شده‌ی تابع مولد دنباله‌ی زیر را بیابید.
    \begin{center}
        $2, 0, 2, 0, 2, ...$
    \end{center}
     
    \problemsolution{
        \p
        اگر 
        $G(X)$
        تابع مولد باشد، داریم:
        \begin{center}
            $G(x)= 2(1) + 0(x) + 2(x^{2}) + 0(x^{3}) + 2(x^{4}) + ... = 2 + 2x^{2} + 2x^{4} + ...$
            
            $G(x)= \sum\limits_{n=0}^{+\infty} 2x^{2n} = 
            2\sum\limits_{n=0}^{+\infty} x^{2n}$
        \end{center}
        با توجه به این که فرم بسته‌ی این تابع 
        $G(x)$
        به دست آمده را به راحتی نمی‌توانیم به دست بیاوریم، سعی می‌کنیم با تغییر تابع مولدی که فرم بسته‌ی آن را می‌دانیم به نتیجه برسیم.
        \begin{center}
            $F(x)= 1 + x + x^{2} + ... = \frac{1}{1-x}$
            
            $F(x^{2})= 1 + x^{2} + x^{4} + ... = \frac{1}{1-x^{2}}$
            
            $G(x) = 2F(x^{2})= 2 + 2x^{2} + 2x^{4} + ... = \frac{2}{1-x^{2}}$
        \end{center}
    }
\end{problem}
