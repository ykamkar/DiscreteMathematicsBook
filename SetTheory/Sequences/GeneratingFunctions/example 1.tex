\begin{problem}
    اگر 
    $a_{n}= {m \choose n}$
    که 
    $n= 0, 1, 2, ..., m$
    باشد، تابع مولد دنباله‌ی 
    $a_{0}, a_{1}, a_{2}, ..., a_{m}$
    چیست؟    
     
    \problemsolution{
        تابع مولد برای این دامنه با توجه به تعریف 
        $a_{n}$
        برابر مقدار زیر می‌شود:
        \begin{center}
            $G(x)= {m \choose 0} + {m \choose 1}x + {m \choose 2}x^{2} + ... + {m \choose m}x^{m}$
        \end{center}
        پس طبق عکس قضیه‌ی ضرایب دو‌جمله‌ای داریم:
        $$G(x)= (1+x)^{m}$$
    }
\end{problem}
