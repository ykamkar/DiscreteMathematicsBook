\begin{problem}
اگر
$a^2, b^2, c^2$
جملات متوالی یک دنباله حسابی باشند، ثابت کنید
$\frac{1}{a + b}, \frac{1}{a + c}, \frac{1}{b + c}$
نیز جملات متوالی یک دنباله حسابی دیگر هستند.
    \problemsolution{
    \begin{center}
        \p
       برای سه جمله متوالی یک دنباله حسابی داریم:
        \p
        $a_{n-1} = \frac{a_n + a_{n-2}}{2}$
        \p
        زیرا:
        \p
        $\frac{a_n + a_{n-2}}{2} = \frac{(a_{n - 1} + d) + (a_{n-1} - d)}{2} = a_{n-1}$
        \p
        پس می‌توان نوشت:
        \p
        $a^2 + c^2 = 2b^2$
        \p
        حال باید ثابت کنیم:
        \p
        $\frac{1}{a + b} + \frac{1}{b + c} = \frac{2}{a + c}$
        \p
        $(2b + a + c)(a + c) = 2(a + b)(b + c)$
        \p
        $\Downarrow$
        \p
        $a^2 + c^2 = 2b^2$
    \end{center}
    }
\end{problem}
