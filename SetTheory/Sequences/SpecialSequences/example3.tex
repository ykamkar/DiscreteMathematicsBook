\begin{problem}
	\p
دنباله‌ای بنویسید که اعضایش فقط از قوانین زیر بدست بیاید.
\begin{enumerate}
	\item
	عضو اول آن
$a_1$
	است.
	\item
$\forall k \in B : (kq) \in B$
\end{enumerate}
    \problemsolution{
        $$a_1 = a_1$$
        $$a_2 = a_1q$$
        $$a_3 = a_2q = (a_1q)q = a_1q^2$$
        $$a_4 = a_1q^4$$
        $$\vdots$$
        $$a_n = a_1q^{n-1}$$
        $$\Downarrow$$
        $$S_n = a_1 + a_2 + \ldots + a_n = a_1(1 + q + q^2 + \ldots + q^{n-1})$$
        با استفاده از اتحاد چاق و لاغر داریم:
        $$Sn = \frac{a_1(1 - q^n)}{1 - q}, (q \neq 1)$$
    }
    
\end{problem}
