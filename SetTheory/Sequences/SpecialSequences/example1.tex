\begin{problem}
    \p
    در یک دنباله حسابی 
    $f_4 + f_{11} = 105$
    و 
    $f_3 + f_7 = 80$
    است.
    $f_1$
    و قدر نسبت را محاسبه کنید.    
     
    \problemsolution{
        \p
        هر مثلث نسبت به مثلث قبلی یک ردیف بیشتر دارد یعنی: 
        \p
        $f_4 + f_{11} = 105$
        \p
        $2f_1 + (3 + 10)d = 105$
        \p
        $f_3 + f_7 = 80$
        \p
        $2f_1 + (2 + 6)d = 80$
        \p
        \begin{center}
        \p
            $d = 5$
            \p
            $f_1 = 20$
        \end{center}
    }
\end{problem}
